\intobmk\chapter*{符号与缩写列表}% 显示在书签但不显示在目录

\section*{符号}
\nomenclatureitem{\textbf{符号}}{\textbf{描述}}
\nomenclatureitem{$I$}{武器集合，$i$代表武器的下标}
\nomenclatureitem{$J$}{雷达集合，$j$代表雷达的下标}
\nomenclatureitem{$K$}{目标集合，$k$代表目标的下标}
\nomenclatureitem{$n$}{目标的数量。}
\nomenclatureitem{$m$}{武器类型的数量。如果每种类型只限定一件，这个变量代表的是武器的数量。}

\nomenclatureitem{$l_i$}{武器容量上限。}
\nomenclatureitem{$L_i$}{武器容量下限（若存在）。}
\nomenclatureitem{$r_j$}{雷达容量上限。}
\nomenclatureitem{$R_j$}{雷达容量下限（若存在）。}

\nomenclatureitem{$V_k$}{目标k的价值。}
\nomenclatureitem{$p_{ik}$}{武器i命中目标k的概率。}
\nomenclatureitem{$p_{ijk}$}{使用武器i和雷达j命中目标k的概率。($x_{ijk}, c_{ijk}$类似)}
\nomenclatureitem{$p_{ijt}^k$}{使用武器i和雷达j在时间点t命中目标k的概率。($x_{ijt}^k, c_{ijt}^k$类似)}
\nomenclatureitem{$x_{ik}$}{分配给目标k的类型为i的武器的数量，可能退化为0-1变量}
\nomenclatureitem{$c_{ik}$}{将武器i分配给目标k的成本。}
